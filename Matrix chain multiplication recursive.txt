#include<bits/stdc++.h>
using namespace std;
int solve(int *a, int i, int j)
{
    if(i>=j)
        return 0;
    int mn=INT_MAX;
    for(int k=i ; k<=j-1 ; k++)
    {
        int tempans=solve(a,i,k)+solve(a,k+1,j)+a[i-1]*a[k]*a[j];
        if(tempans<mn)
            mn=tempans;
    }
    return mn;
}
int main()
{
    int n;
    cout<<"Enter the size of aaray : ";
    cin>>n;
    int *a = new int[n];
    cout<<"Enter the array : ";
    for(int i=0 ; i<n ; i++)
        cin>>a[i];
    cout<<"Matrix Chain Multiplication is :"<<solve(a,1,n-1);
    return 0;
}
