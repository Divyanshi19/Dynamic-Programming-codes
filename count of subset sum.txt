#include<iostream>
using namespace std;
int knapsack(int *w, int n, int sack)
{
    int dp[n+1][sack+1];
   for (int i = 0; i <= n; i++) 
      dp[i][0] = true; 
    for (int i = 1; i <= sack; i++) 
      dp[0][i] = false; 
    for(int i=1 ; i<n+1 ; i++)
    {
        for(int j=1 ; j<sack+1 ; j++)
        {
            
            if(w[i-1] > j)
                dp[i][j]=dp[i-1][j];
            else
                dp[i][j]=dp[i-1][j] + dp[i-1][j-w[i-1]];
        }

    }
    return dp[n][sack];
}
int main()
{
    int n,knap;
    cout<<"Enter the no. of operations"<<endl;
    cin>>n;
    int *weight = new int[n];
    //*value=new int[n];
    cout<<"Enter the weight "<<endl;
    for(int i=0 ; i<n ; i++)
    {
        cin>>weight[i];
        //value[i];
    }
    cout<<"Enter the sum"<<endl;
    cin>>knap;
    cout<<"Value: "<<knapsack(weight,n,knap)<<endl;
}
